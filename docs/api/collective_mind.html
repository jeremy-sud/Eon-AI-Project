 59class TrueWillVector:
 60    """
 61    Vector de Voluntad Verdadera (Thelema).
 62    
 63    Cada nodo desarrolla una "Voluntad" basada en:
 64    - Lo que nació procesando (genesis domain)
 65    - Lo que ha aprendido mejor (experiencia)
 66    - Su inercia hacia ciertos tipos de datos
 67    
 68    La Voluntad determina qué tareas acepta/rechaza.
 69    
 70    Filosofía:
 71    ---------
 72    "Cada estrella tiene su órbita. No hay colisión si cada una
 73    sigue su camino. La fricción del sistema se minimiza cuando
 74    cada nodo hace SOLO lo que nació para hacer."
 75    
 76    Implementación:
 77    --------------
 78    - affinity[domain] ∈ [0, 1]: Afinidad hacia un dominio
 79    - inertia: Resistencia a cambiar de dominio
 80    - cost_function: Penalización por tareas fuera de voluntad
 81    """
 82    
 83    def __init__(self, genesis_domain: DataDomain = DataDomain.GENERIC):
 84        """
 85        Args:
 86            genesis_domain: Dominio inicial/nativo del nodo
 87        """
 88        self.genesis_domain = genesis_domain
 89        
 90        # Afinidad por cada dominio [0-1]
 91        # El dominio genesis comienza con afinidad máxima
 92        self.affinity: Dict[DataDomain, float] = {
 93            domain: 0.1 for domain in DataDomain
 94        }
 95        self.affinity[genesis_domain] = 1.0
 96        
 97        # Inercia: resistencia al cambio (aumenta con experiencia)
 98        self.inertia: float = 0.5
 99        
100        # Historial de procesamiento por dominio
101        self.processing_history: Dict[DataDomain, int] = {
102            domain: 0 for domain in DataDomain
103        }
104        self.processing_history[genesis_domain] = 1
105        
106        # Métricas de éxito por dominio (MSE promedio)
107        self.success_metrics: Dict[DataDomain, List[float]] = {
108            domain: [] for domain in DataDomain
109        }
110        
111        # Umbrales de rechazo
112        self.rejection_threshold: float = 0.3  # Rechaza si afinidad < esto
113        self.high_cost_threshold: float = 0.5  # Costo alto si afinidad < esto
114    
115    def calculate_true_will_vector(self) -> Dict[str, float]:
116        """
117        Calcula el vector de Voluntad Verdadera normalizado.
118        
119        Retorna un diccionario con la "fuerza de voluntad" hacia cada dominio.
120        La suma de todos los valores = 1.0
121        
122        Returns:
123            Dict[domain_name, will_strength]
124        """
125        # Combinar afinidad con experiencia
126        will_vector = {}
127        total = 0.0
128        
129        for domain in DataDomain:
130            # Experiencia normalizada
131            total_processing = sum(self.processing_history.values())
132            experience = self.processing_history[domain] / max(1, total_processing)
133            
134            # Éxito promedio (menor MSE = mayor éxito)
135            if self.success_metrics[domain]:
136                avg_mse = np.mean(self.success_metrics[domain][-10:])  # Últimos 10
137                success = 1.0 / (1.0 + avg_mse)  # Inversamente proporcional
138            else:
139                success = 0.5 if domain == self.genesis_domain else 0.1
140            
141            # Fórmula de Voluntad: afinidad × (1 + experiencia) × éxito
142            will = self.affinity[domain] * (1 + experience) * success
143            will_vector[domain.value] = will
144            total += will
145        
146        # Normalizar
147        if total > 0:
148            for key in will_vector:
149                will_vector[key] /= total
150        
151        return will_vector
152    
153    def evaluate_task_cost(self, requested_domain: DataDomain) -> Tuple[float, str]:
154        """
155        Evalúa el costo de procesar una tarea en un dominio específico.
156        
157        Retorna:
158        - cost: 0.0 (alineado) a 1.0 (máxima resistencia)
159        - decision: "accept", "high_priority", "low_priority", "reject"
160        
161        Args:
162            requested_domain: Dominio de la tarea solicitada
163            
164        Returns:
165            (cost, decision)
166        """
167        affinity = self.affinity[requested_domain]
168        
169        # Costo base inversamente proporcional a afinidad
170        base_cost = 1.0 - affinity
171        
172        # Modificar por inercia (nodos con mucha inercia resisten más)
173        adjusted_cost = base_cost * (1 + self.inertia)
174        adjusted_cost = min(1.0, adjusted_cost)
175        
176        # Determinar decisión
177        if affinity >= self.high_cost_threshold:
178            decision = "accept" if affinity >= 0.8 else "high_priority"
179        elif affinity >= self.rejection_threshold:
180            decision = "low_priority"
181        else:
182            decision = "reject"
183        
184        return adjusted_cost, decision
185    
186    def should_accept_task(self, requested_domain: DataDomain) -> bool:
187        """
188        Decisión binaria: ¿debería este nodo aceptar esta tarea?
189        
190        Basado en el principio Thelemático: "Cada estrella en su órbita."
191        Un nodo no debe desviarse de su Voluntad Verdadera.
192        """
193        _, decision = self.evaluate_task_cost(requested_domain)
194        return decision != "reject"
195    
196    def record_processing(self, domain: DataDomain, mse: float) -> None:
197        """
198        Registra el procesamiento de datos en un dominio.
199        
200        Esto actualiza la afinidad y experiencia del nodo.
201        
202        Args:
203            domain: Dominio procesado
204            mse: Error cuadrático medio del procesamiento
205        """
206        # Incrementar experiencia
207        self.processing_history[domain] += 1
208        
209        # Registrar métrica de éxito
210        self.success_metrics[domain].append(mse)
211        
212        # Actualizar afinidad basada en éxito
213        if mse < 0.1:  # Muy exitoso
214            self.affinity[domain] = min(1.0, self.affinity[domain] + 0.05)
215        elif mse < 0.3:  # Aceptable
216            self.affinity[domain] = min(1.0, self.affinity[domain] + 0.02)
217        elif mse > 0.7:  # Malo - reducir afinidad
218            self.affinity[domain] = max(0.0, self.affinity[domain] - 0.03)
219        
220        # Aumentar inercia con experiencia total
221        total_exp = sum(self.processing_history.values())
222        self.inertia = min(0.95, 0.5 + (total_exp / 1000))
223    
224    def get_specialization(self) -> Tuple[DataDomain, float]:
225        """
226        Retorna el dominio de especialización del nodo y su nivel.
227        
228        Returns:
229            (domain, specialization_level)
230        """
231        max_affinity = 0.0
232        specialized_domain = self.genesis_domain
233        
234        for domain, affinity in self.affinity.items():
235            if affinity > max_affinity:
236                max_affinity = affinity
237                specialized_domain = domain
238        
239        return specialized_domain, max_affinity
240    
241    def export_will(self) -> Dict:
242        """Exporta el vector de voluntad para sincronización."""
243        return {
244            'genesis_domain': self.genesis_domain.value,
245            'affinity': {d.value: v for d, v in self.affinity.items()},
246            'inertia': self.inertia,
247            'will_vector': self.calculate_true_will_vector(),
248            'specialization': {
249                'domain': self.get_specialization()[0].value,
250                'level': self.get_specialization()[1]
251            },
252            'processing_history': {d.value: v for d, v in self.processing_history.items()}
253        }

256class AeonNode:
257    """
258    Un nodo de la Mente Colectiva.
259    
260    Cada nodo puede:
261    - Aprender localmente
262    - Exportar sus pesos W_out
263    - Importar pesos de otros nodos
264    - Mezclar conocimiento (promedio ponderado)
265    - Calcular su Voluntad Verdadera (Thelema)
266    - Aceptar/rechazar tareas según su especialización
267    """

269    def __init__(self, node_id: str = None, n_reservoir: int = 32,
270                 genesis_domain: DataDomain = DataDomain.GENERIC):
271        """
272        Args:
273            node_id: Identificador único (auto-generado si None)
274            n_reservoir: Tamaño del reservoir
275            genesis_domain: Dominio nativo del nodo (para Thelema)
276        """
277        self.node_id = node_id or self._generate_id()
278        self.n_reservoir = n_reservoir
279        
280        self.esn = EchoStateNetwork(
281            n_inputs=1,
282            n_outputs=1,
283            n_reservoir=n_reservoir,
284            spectral_radius=0.9,
285            sparsity=0.85
286        )
287        
288        self.created_at = datetime.now(timezone.utc).isoformat()
289        self.samples_learned = 0
290        self.sync_count = 0
291        self.last_sync = None
292        self.peers: List[str] = []
293        
294        # Sistema de Voluntad Verdadera (Thelema)
295        self.true_will = TrueWillVector(genesis_domain)

297    def _generate_id(self) -> str:
298        """Genera ID único basado en timestamp + random."""
299        rng = np.random.default_rng(int(time.time() * 1000) % (2**32))
300        data = f"{time.time()}-{rng.random()}"
301        return hashlib.sha256(data.encode()).hexdigest()[:12]

303    def train(self, data: np.ndarray, washout: int = 20,
304              domain: DataDomain = DataDomain.TIMESERIES) -> float:
305        """
306        Entrena localmente con datos.
307        
308        Args:
309            data: Serie temporal 1D
310            washout: Muestras a descartar
311            domain: Dominio de los datos (para Thelema)
312            
313        Returns:
314            MSE del entrenamiento
315        """
316        X = data[:-1].reshape(-1, 1)
317        Y = data[1:].reshape(-1, 1)
318        
319        self.esn.fit(X, Y, washout=washout)
320        self.samples_learned += len(data)
321        
322        # Calcular MSE
323        predictions = self.esn.predict(X)
324        mse = np.mean((predictions - Y) ** 2)
325        
326        # Registrar en sistema Thelema
327        self.true_will.record_processing(domain, float(mse))
328        
329        return float(mse)

331    def evaluate_task(self, domain: DataDomain) -> Tuple[bool, float, str]:
332        """
333        Evalúa si este nodo debería aceptar una tarea en un dominio específico.
334        
335        Implementa el principio de "Voluntad Verdadera" (Thelema):
336        - Si la tarea está alineada con la Voluntad del nodo → acepta
337        - Si está desalineada → rechaza o baja prioridad
338        
339        Args:
340            domain: Dominio de la tarea solicitada
341            
342        Returns:
343            (should_accept, cost, decision)
344        """
345        cost, decision = self.true_will.evaluate_task_cost(domain)
346        should_accept = self.true_will.should_accept_task(domain)
347        return should_accept, cost, decision

355        
356        Returns:
357            Dict con fuerza de voluntad por dominio (suma = 1.0)

386    def import_weights(self, weights_data: Dict, merge_ratio: float = 0.5) -> bool:
387        """
388        Importa y mezcla pesos de otro nodo.
389        
390        Args:
391            weights_data: dict del export_weights de otro nodo
392            merge_ratio: Peso del conocimiento externo (0-1)
393                        0 = ignorar externo, 1 = reemplazar totalmente
394        
395        Returns:
396            True si éxito
397        """
398        if weights_data['n_reservoir'] != self.n_reservoir:
399            raise ValueError(f"Tamaño de reservoir incompatible: "
400                           f"{weights_data['n_reservoir']} vs {self.n_reservoir}")
401        
402        external_weights = np.array(weights_data['W_out'])
403        
404        # Mezclar: local * (1-ratio) + externo * ratio
405        self.esn.W_out = (
406            self.esn.W_out * (1 - merge_ratio) +
407            external_weights * merge_ratio
408        )
409        
410        self.sync_count += 1
411        self.last_sync = datetime.now(timezone.utc).isoformat()
412        
413        if weights_data['node_id'] not in self.peers:
414            self.peers.append(weights_data['node_id'])
415        
416        return True

440    def export_weights_1bit(self, scale: float = 0.5) -> Dict:
441        """
442        Exporta pesos W_out con cuantización 1-bit (Protocolo Eón).
443        
444        Solo se transmite el signo de cada peso, logrando compresión 17x.
445        
446        Args:
447            scale: Magnitud para reconstrucción (default: 0.5)
448            
449        Returns:
450            dict con metadatos y payload binario
451        """
452        w_out = self.esn.W_out.flatten()
453        
454        # Cuantizar a 1-bit (solo signos)
455        n_weights = len(w_out)
456        n_bytes = (n_weights + 7) // 8
457        payload = bytearray(n_bytes)
458        
459        for i, w in enumerate(w_out):
460            if w >= 0:
461                byte_idx = i // 8
462                bit_idx = i % 8
463                payload[byte_idx] |= (1 << bit_idx)
464        
465        # Construir paquete según protocolo
466        return {
467            'magic': 'EON',
468            'type': 0x01,  # W_OUT_UPDATE
469            'seed': hash(self.node_id) & 0xFFFFFFFF,
470            'count': n_weights,
471            'scale': scale,
472            'payload': bytes(payload),
473            'node_id': self.node_id,
474            'samples_learned': self.samples_learned,
475            'timestamp': datetime.now(timezone.utc).isoformat(),
476            # Estadísticas de compresión
477            'original_bytes': n_weights * 4,  # float32
478            'compressed_bytes': n_bytes + 10,  # payload + header
479            'compression_ratio': (n_weights * 4) / (n_bytes + 10)
480        }

482    def import_weights_1bit(self, packet: Dict, merge_ratio: float = 0.5) -> bool:
483        """
484        Importa pesos cuantizados 1-bit y los mezcla con los locales.
485        
486        Args:
487            packet: Paquete del export_weights_1bit
488            merge_ratio: Peso del conocimiento externo (0-1)
489            
490        Returns:
491            True si éxito
492        """
493        if packet.get('magic') != 'EON':
494            raise ValueError("Paquete inválido: magic incorrecto")
495        
496        n_weights = packet['count']
497        scale = packet.get('scale', 0.5)
498        payload = packet['payload']
499        
500        # Verificar tamaño
501        expected_bytes = (n_weights + 7) // 8
502        if len(payload) != expected_bytes:
503            raise ValueError(f"Payload size mismatch: {len(payload)} vs {expected_bytes}")
504        
505        # Dequantizar
506        external_weights = np.zeros(n_weights)
507        for i in range(n_weights):
508            byte_idx = i // 8
509            bit_idx = i % 8
510            if payload[byte_idx] & (1 << bit_idx):
511                external_weights[i] = scale
512            else:
513                external_weights[i] = -scale
514        
515        # Reshape para coincidir con W_out
516        external_weights = external_weights.reshape(self.esn.W_out.shape)
517        
518        # Mezclar
519        self.esn.W_out = (
520            self.esn.W_out * (1 - merge_ratio) +
521            external_weights * merge_ratio
522        )
523        
524        self.sync_count += 1
525        self.last_sync = datetime.now(timezone.utc).isoformat()
526        
527        if packet.get('node_id') and packet['node_id'] not in self.peers:
528            self.peers.append(packet['node_id'])
529        
530        return True

580    def apply_egregore_recommendations(self, state: EgregorState) -> Dict:
581        """
582        Aplica las recomendaciones homeostáticas del Egrégor.
583        
584        Implementa el feedback loop: cuando el Egrégor está "Agitado",
585        el nodo reduce su actividad para "calmar" al sistema.
586        
587        Args:
588            state: Estado actual del Egrégor
589            
590        Returns:
591            Dict con las acciones tomadas
592        """
593        actions_taken = {
594            "sample_rate_adjusted": False,
595            "merge_ratio_adjusted": False,
596            "activity_reduced": False,
597            "activity_increased": False,
598        }
599        
600        recommendations = state.get_homeostatic_actions()
601
