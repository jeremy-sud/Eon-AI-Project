 65    def _generate_hash(self, seed: int, timestamp: int) -> str:
 66        """Genera hash de nacimiento estandarizado (compatible con C/JS)."""
 67        state = seed ^ timestamp
 68        # LCG constants from C implementation (mixed/simplified)
 69        # Using a simple Python implementation to mimic the C logic 
 70        # C: state = (state * 1103515245 + 12345) & 0x7fffffff
 71        
 72        bytes_list = []
 73        curr = state & 0xFFFFFFFF
 74        for _ in range(16):
 75            curr = (curr * 1103515245 + 12345) & 0x7fffffff
 76            bytes_list.append(curr & 0xFF)
 77            
 78        return ''.join(f'{b:02x}' for b in bytes_list)

 80    def _initialize_weights(self):
 81        """Inicializa las matrices de pesos aleatorios."""
 82        
 83        # Matriz de entrada: conexiones aleatorias input -> reservoir
 84        self.W_in = self.rng.uniform(-1, 1, (self.n_reservoir, self.n_inputs))
 85        
 86        # Matriz del reservoir: conexiones recurrentes
 87        # Aplicamos escasez (sparsity) para eficiencia
 88        W = self.rng.uniform(-1, 1, (self.n_reservoir, self.n_reservoir))
 89        mask = self.rng.random((self.n_reservoir, self.n_reservoir)) > self.sparsity
 90        W *= mask
 91        
 92        # Escalar al radio espectral deseado para estabilidad
 93        # El eigenvalor máximo controla la "memoria" del sistema
 94        eigenvalues = np.abs(np.linalg.eigvals(W))
 95        if eigenvalues.max() > 0:
 96            W *= self.spectral_radius / eigenvalues.max()
 97        
 98        self.W_reservoir = W

100    def _update_state(self, input_vector: np.ndarray) -> np.ndarray:
101        """
102        Actualiza el estado del reservoir dado un input.
103        
104        Ecuación: state(t+1) = tanh(W_in * input + W_reservoir * state(t) + noise)
105        
106        Args:
107            input_vector: Vector de entrada
108            
109        Returns:
110            Nuevo estado del reservoir
111        """
112        # Proyección del input
113        input_contribution = np.dot(self.W_in, input_vector)
114        
115        # Recurrencia del reservoir
116        reservoir_contribution = np.dot(self.W_reservoir, self.state)
117        
118        # Ruido para regularización
119        noise_contribution = self.noise * self.rng.standard_normal(self.n_reservoir)
120        
121        # Nuevo estado con no-linealidad tanh
122        self.state = np.tanh(input_contribution + reservoir_contribution + noise_contribution)
123        
124        return self.state

126    def fit(self, inputs: np.ndarray, outputs: np.ndarray, washout: int = 100) -> 'EchoStateNetwork':
127        """
128        Entrena SOLO la capa de salida mediante regresión lineal.
129        
130        El reservoir NUNCA se modifica. Esto es "La Nada es Todo":
131        el reservoir aleatorio ya contiene toda la representación necesaria,
132        solo necesitamos aprender a leerla.
133        
134        Args:
135            inputs: Secuencia de entrada (T, n_inputs)
136            outputs: Secuencia objetivo (T, n_outputs)
137            washout: Pasos iniciales a descartar (para que el estado converja)
138            
139        Returns:
140            self (para encadenamiento)
141        """
142        T = inputs.shape[0]
143        
144        # Asegurar dimensiones correctas
145        if inputs.ndim == 1:
146            inputs = inputs.reshape(-1, 1)
147        if outputs.ndim == 1:
148            outputs = outputs.reshape(-1, 1)
149            
150        # Recolectar estados del reservoir
151        states = np.zeros((T, self.n_reservoir))
152        
153        # Reset estado inicial
154        self.state = np.zeros(self.n_reservoir)
155        
156        # Pasar todos los inputs por el reservoir
157        for t in range(T):
158            states[t] = self._update_state(inputs[t])
159            
160        # Descartar período de "calentamiento" (washout)
161        states = states[washout:]
162        outputs_train = outputs[washout:]
163        
164        # Regresión lineal con regularización Ridge
165        # W_out = (S^T * S + λI)^(-1) * S^T * Y
166        reg = 1e-6  # Regularización
167        S = states
168        Y = outputs_train
169        
170        self.W_out = np.dot(
171            np.linalg.inv(np.dot(S.T, S) + reg * np.eye(self.n_reservoir)),
172            np.dot(S.T, Y)
173        )
174        
175        return self

def generate_mackey_glass( n_samples: int = 2000, tau: int = 17, delta_t: float = 1.0) -> numpy.ndarray: 
264def generate_mackey_glass(n_samples: int = 2000, tau: int = 17, delta_t: float = 1.0) -> np.ndarray:
265    """
266    Genera la serie temporal Mackey-Glass.
267    
268    Esta es una serie caótica clásica usada para evaluar modelos de predicción.
269    Es determinista pero altamente no-lineal.
270    
271    Args:
272        n_samples: Número de muestras a generar
273        tau: Parámetro de retardo (17 produce caos)
274        delta_t: Paso temporal
275        
276    Returns:
277        Serie temporal Mackey-Glass
278    """
279    # Inicialización
280    history_len = tau
281    x_history = np.ones(history_len) * 1.2
282    x_t = 1.2
283    
284    series = np.zeros(n_samples)
285    
286    for t in range(n_samples):
287        # Ecuación de Mackey-Glass
288        x_tau = x_history[0]
289        dx = delta_t * (0.2 * x_tau / (1.0 + x_tau**10) - 0.1 * x_t)
290        x_t = x_t + dx
291        
292        # Actualizar historia
293        x_history = np.roll(x_history, -1)
294        x_history[-1] = x_t
295        
296        series[t] = x_t
297        
298    return series

162    """
163    Echo State Network Recursivo/Fractal.
164    
165    Implementa el principio hermético "Como es Arriba, es Abajo":
166    cada unidad del reservoir puede ser un micro-reservoir.
167    
168    Arquitectura:
169    - Nivel Superior (Macro): Conexiones entre unidades
170    - Nivel Inferior (Micro): Cada unidad es un mini-ESN
171    
172    Esto permite procesar patrones a múltiples escalas temporales:
173    - Las unidades con time_scale=1 capturan patrones rápidos
174    - Las unidades con time_scale=10 capturan patrones lentos
175    
176    Ventajas:
177    1. Memoria multi-escala natural
178    2. Jerarquía emergente sin supervisión
179    3. Mismo código recursivo para todos los niveles
180    4. Eficiente: solo entrena W_out (como ESN normal)
181    """
182    
183    def __init__(
184        self,
185        n_inputs: int = 1,
186        n_outputs: int = 1,
187        n_macro_units: int = 10,
188        n_micro_neurons: int = 8,
189        time_scales: Optional[List[float]] = None,
190        spectral_radius: float = 0.9,
191        macro_spectral_radius: float = 0.95,
192        sparsity: float = 0.8,
193        macro_sparsity: float = 0.7,
194        noise: float = 0.001,
195        random_state: Optional[int] = None
196    ):
197        """
198        Args:
199            n_inputs: Dimensión de entrada
200            n_outputs: Dimensión de salida
201            n_macro_units: Número de unidades macro (cada una es un micro-ESN)
202            n_micro_neurons: Neuronas por micro-ESN
203            time_scales: Escalas temporales para cada unidad (None = automático)
204            spectral_radius: Radio espectral de micro-reservoirs
205            macro_spectral_radius: Radio espectral de conexiones macro
206            sparsity: Escasez en micro-reservoirs
207            macro_sparsity: Escasez en conexiones macro
208            noise: Nivel de ruido
209            random_state: Semilla aleatoria
210        """
211        self.n_inputs = n_inputs
212        self.n_outputs = n_outputs
213        self.n_macro_units = n_macro_units
214        self.n_micro_neurons = n_micro_neurons
215        self.noise = noise
216        
217        # Generador aleatorio (API moderno)
218        self.rng = np.random.default_rng(random_state)
219        
220        # === NIVEL MICRO (Abajo) ===
221        # Crear micro-reservoirs con diferentes escalas temporales
222        if time_scales is None:
223            # Distribución logarítmica de escalas (1, 2, 4, 8, ...)
224            time_scales = [2 ** (i * 0.5) for i in range(n_macro_units)]
225        
226        self.time_scales = time_scales[:n_macro_units]
227        
228        self.micro_reservoirs: List[MicroReservoir] = []
229        for i in range(n_macro_units):
230            scale = self.time_scales[i] if i < len(self.time_scales) else 1.0
231            micro = MicroReservoir(

297        """

Echo State Network Recursivo/Fractal.

Implementa el principio hermético "Como es Arriba, es Abajo": cada unidad del reservoir puede ser un micro-reservoir.

Arquitectura:

  • Nivel Superior (Macro): Conexiones entre unidades
  • Nivel Inferior (Micro): Cada unidad es un mini-ESN

Esto permite procesar patrones a múltiples escalas temporales:

  • Las unidades con time_scale=1 capturan patrones rápidos
  • Las unidades con time_scale=10 capturan patrones lentos

Ventajas:

  1. Memoria multi-escala natural
  2. Jerarquía emergente sin supervisión
  3. Mismo código recursivo para todos los niveles
  4. Eficiente: solo entrena W_out (como ESN normal)

 79        self.n_internal = n_internal
 80        self.time_scale = time_scale
 81        self.spectral_radius = spectral_radius
 82        self.sparsity = sparsity
 83        self.rng = rng if rng is not None else np.random.default_rng()

104        # Permite que este micro-reservoir "piense más lento"

110        """
111        Actualiza el estado del micro-reservoir.
112        
113        Si time_scale > 1, acumula inputs antes de actualizar

115        
116        Args:
117            input_vec: Vector de entrada
118            noise: Nivel de ruido
119            
120        Returns:
121            Estado actual (puede no cambiar si está acumulando)
122        """
123        # Acumular input para escala temporal
124        self._accumulated_input += input_vec
125        self._accumulation_count += 1
126

 61        self,
 62        n_inputs: int,
 63        n_internal: int = 8,
 64        time_scale: float = 1.0,
 65        spectral_radius: float = 0.9,
 66        sparsity: float = 0.8,
 67        rng: Optional[np.random.Generator] = None
 68    ):
 69        """
 70        Args:
 71            n_inputs: Dimensión de entrada
 72            n_internal: Neuronas internas del micro-reservoir
 73            time_scale: Escala temporal (mayor = más lento, más memoria)
 74            spectral_radius: Radio espectral
 75            sparsity: Escasez de conexiones
 76            rng: Generador aleatorio (np.random.Generator)
 77        """
 78        self.n_inputs = n_inputs
 79        self.n_internal = n_internal
 80        self.time_scale = time_scale
 81        self.spectral_radius = spectral_radius
 82        self.sparsity = sparsity
 83        self.rng = rng if rng is not None else np.random.default_rng()
 84        
 85        # Pesos internos
 86        self.W_in = self.rng.uniform(-1, 1, (n_internal, n_inputs))
 87        
 88        # Matriz recurrente interna
 89        W = self.rng.uniform(-1, 1, (n_internal, n_internal))
 90        mask = self.rng.random((n_internal, n_internal)) > sparsity
 91        W *= mask
 92        
 93        # Escalar al radio espectral
 94        eigenvalues = np.abs(np.linalg.eigvals(W))
 95        if eigenvalues.max() > 0:
 96            W *= spectral_radius / eigenvalues.max()
 97        
 98        self.W_internal = W
 99        
100        # Estado interno
101        self.state = np.zeros(n_internal)
102        
103        # Acumulador para escala temporal
104        # Permite que este micro-reservoir "piense más lento"
105        self._accumulated_input = np.zeros(n_inputs)
106        self._accumulation_count = 0
107        self._update_threshold = max(1, int(time_scale))
