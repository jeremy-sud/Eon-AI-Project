 28class WordTokenizer:
 29    """
 30    Tokenizador a nivel de palabra con embeddings simples.
 31    """
 32    
 33    def __init__(self, vocab_size: int = 500, embedding_dim: int = 32):
 34        self.vocab_size = vocab_size
 35        self.embedding_dim = embedding_dim
 36        # Replace dicts with TrieVocab
 37        self.vocab = TrieVocab()
 38        self.embeddings: Optional[np.ndarray] = None
 39        
 40        # Tokens especiales
 41        self.PAD = "<PAD>"
 42        self.UNK = "<UNK>"
 43        self.BOS = "<BOS>"
 44        self.EOS = "<EOS>"
 45        
 46        # Pre-seed special tokens
 47        self.vocab.add(self.PAD)
 48        self.vocab.add(self.UNK)
 49        self.vocab.add(self.BOS)
 50        self.vocab.add(self.EOS)
 51        
 52    def fit(self, text: str):
 53        """Construye vocabulario y embeddings."""
 54        # Tokenizar
 55        words = self._tokenize(text)
 56        
 57        # Contar frecuencias
 58        counter = Counter(words)
 59        
 60        # Agregar palabras más frecuentes al Trie
 61        # Note: Trie grows sequentially, so ID order corresponds to insertion order (roughly)
 62        # We need to make sure we don't exceed desired size if possible, or just add all.
 63        # Ideally we add most frequent first.
 64        
 65        target_size = self.vocab_size - 4
 66        for w, _ in counter.most_common(target_size):
 67            self.vocab.add(w)
 68            
 69        self.actual_vocab_size = self.vocab.size
 70        
 71        # Crear embeddings aleatorios
 72        rng = np.random.default_rng(42)
 73        self.embeddings = rng.standard_normal((self.actual_vocab_size, self.embedding_dim)) * 0.1
 74        self.embeddings[0] = 0 # PAD
 75        
 76    def _tokenize(self, text: str) -> List[str]:
 77        """Tokeniza texto en palabras."""
 78        # Normalizar
 79        text = text.lower()
 80        # Separar puntuación
 81        for p in '.,!?;:\n':
 82            text = text.replace(p, f' {p} ')
 83        return text.split()
 84    
 85    def encode(self, text: str) -> np.ndarray:
 86        """Convierte texto a índices."""
 87        words = self._tokenize(text)
 88        unk_idx = self.vocab.get_id(self.UNK)
 89        
 90        indices = []
 91        for w in words:
 92            idx = self.vocab.get_id(w)
 93            if idx == -1:
 94                indices.append(unk_idx)
 95            else:
 96                indices.append(idx)
 97        return np.array(indices)
 98    
 99    def decode(self, indices: np.ndarray) -> str:
100        """Convierte índices a texto."""
101        words = []
102        for i in indices:
103            w = self.vocab.get_word(int(i))
104            if w == "<UNK>" and int(i) != self.vocab.get_id(self.UNK):
105                # Should not happen if within bounds
106                pass
107            words.append(w)
108        # Limpiar tokens especiales
109        words = [w for w in words if w not in [self.PAD, self.BOS, self.EOS]]
110        # Reconstruir texto
111        text = ' '.join(words)
112        # Limpiar espacios antes de puntuación
113        for p in '.,!?;:':
114            text = text.replace(f' {p}', p)
115        return text
116    
117    def to_embeddings(self, indices: np.ndarray) -> np.ndarray:
118        """Convierte índices a embeddings."""
119        return self.embeddings[indices]

122class TinyLMv2:
123    """
124    Modelo de Lenguaje ESN Mejorado.
125    
126    Características:
127    - Tokenización por palabras
128    - Embeddings densos o gemátricos
129    - Reservoir grande
130    - Múltiples estrategias de decodificación
131    """
132    
133    def __init__(self, 
134                 n_reservoir: int = 256,
135                 vocab_size: int = 500,
136                 embedding_dim: int = 32,
137                 tokenizer_type: Literal["word", "gematria"] = "word"):
138        """
139        Inicializa el modelo de lenguaje.
140        
141        Args:
142            n_reservoir: Neuronas en el reservoir
143            vocab_size: Tamaño máximo del vocabulario
144            embedding_dim: Dimensión de embeddings
145            tokenizer_type: Tipo de tokenizador ("word" o "gematria")
146        """
147        self.n_reservoir = n_reservoir
148        self.vocab_size = vocab_size
149        self.embedding_dim = embedding_dim
150        self.tokenizer_type = tokenizer_type
151        
152        # Seleccionar tokenizador según tipo
153        if tokenizer_type == "gematria":
154            # 93 buckets = valor de Thelema en gematria
155            self.tokenizer = GematriaTokenizer(
156                embedding_dim=embedding_dim,
157                n_buckets=93,
158                min_freq=1
159            )
160        else:
161            self.tokenizer = WordTokenizer(vocab_size, embedding_dim)
162        
163        self.esn = None
164        self.is_trained = False
165        
166        # Matriz de proyección de salida (reservoir -> vocab)
167        self.output_projection = None
168        
169    def train(self, text: str, epochs: int = 3, washout: int = 50) -> Dict:
170        """
171        Entrena el modelo.
172        
173        Args:
174            text: Texto de entrenamiento
175            epochs: Pasadas sobre los datos
176            washout: Muestras iniciales a descartar
177            
178        Returns:
179            Estadísticas de entrenamiento
180        """
181        # Construir vocabulario
182        self.tokenizer.fit(text)
183        
184        # Crear ESN
185        self.esn = EchoStateNetwork(
186            n_inputs=self.embedding_dim,
187            n_outputs=self.tokenizer.actual_vocab_size,
188            n_reservoir=self.n_reservoir,
189            spectral_radius=0.95,
190            sparsity=0.9,
191            noise=0.0001
192        )
193        
194        # Preparar datos
195        indices = self.tokenizer.encode(text)
196        
197        if len(indices) < washout + 10:
198            raise ValueError("Texto demasiado corto")
199        
200        # Múltiples épocas: repetir el texto
201        all_indices = np.tile(indices, epochs)
202        
203        # Embeddings
204        x_emb = self.tokenizer.to_embeddings(all_indices[:-1])
205        
206        # Targets: one-hot del siguiente token
207        Y = np.zeros((len(all_indices) - 1, self.tokenizer.actual_vocab_size))
208        for i, idx in enumerate(all_indices[1:]):
209            Y[i, idx] = 1
210        
211        # Entrenar
212        self.esn.fit(x_emb, Y, washout=washout)
213        self.is_trained = True
214        
215        # Evaluar
216        predictions = self.esn.predict(x_emb)
217        pred_indices = np.argmax(predictions, axis=1)
218        real_indices = all_indices[1:]
219        
220        accuracy = np.mean(pred_indices == real_indices)
221        
222        # Top-5 accuracy
223        top5_correct = 0
224        for i, pred in enumerate(predictions):
225            top5 = np.argsort(pred)[-5:]
226            if real_indices[i] in top5:
227                top5_correct += 1
228        top5_acc = top5_correct / len(predictions)
229        
230        return {
231            'accuracy': float(accuracy),
232            'top5_accuracy': float(top5_acc),
233            'vocab_size': self.tokenizer.actual_vocab_size,
234            'n_reservoir': self.n_reservoir,
235            'total_tokens': len(all_indices)
236        }
237    
238    def _get_embedding(self, idx: int) -> np.ndarray:
239        """Obtiene el embedding para un índice."""
240        if hasattr(self.tokenizer, '_embeddings') and self.tokenizer._embeddings is not None:
241            return self.tokenizer._embeddings[idx]
242        return self.tokenizer.embeddings[idx]
243    
244    def _get_token_id(self, token: str) -> int:
245        """Obtiene el ID de un token especial."""
246        if hasattr(self.tokenizer, 'word2idx'):
247            return self.tokenizer.word2idx.get(token, 0)
248        return self.tokenizer.vocab.get_id(token)
249    
250    def _select_next_greedy(self, logits: np.ndarray) -> int:
251        """Selección greedy del siguiente token."""
252        return int(np.argmax(logits))
253    
254    def _select_next_sampling(
255        self, 
256        logits: np.ndarray, 
257        temperature: float, 
258        top_k: int
259    ) -> int:
260        """Selección por sampling con top-k."""
261        import time
262        
263        # Aplicar temperatura
264        logits = logits / (temperature + 1e-8)
265        
266        # Top-k filtering
267        top_k_indices = np.argsort(logits)[-top_k:]
268        top_k_logits = logits[top_k_indices]
269        
270        # Softmax
271        exp_logits = np.exp(top_k_logits - np.max(top_k_logits))
272        probs = exp_logits / np.sum(exp_logits)
273        
274        # Samplear
275        rng = np.random.default_rng(int(time.time() * 1000) % (2**32))
276        chosen = rng.choice(len(top_k_indices), p=probs)
277        return int(top_k_indices[chosen])
278    
279    def _avoid_special_tokens(self, next_idx: int, logits: np.ndarray) -> int:
280        """Evita seleccionar tokens especiales (PAD, UNK)."""
281        if next_idx in [0, 1]:
282            sorted_indices = np.argsort(logits)[::-1]
283            for idx in sorted_indices:
284                if idx not in [0, 1]:
285                    return int(idx)
286        return next_idx
287    
288    def generate(self, 
289                 prompt: str, 
290                 max_tokens: int = 30,
291                 temperature: float = 0.7,
292                 top_k: int = 10,
293                 strategy: str = 'sampling') -> str:
294        """
295        Genera texto.
296        
297        Args:
298            prompt: Texto inicial
299            max_tokens: Máximo de tokens a generar
300            temperature: Creatividad (menor = más determinista)
301            top_k: Número de candidatos para sampling
302            strategy: 'greedy', 'sampling', o 'beam'
303            
304        Returns:
305            Texto generado
306        """
307        if not self.is_trained:
308            raise RuntimeError("Modelo no entrenado")
309        
310        # Resetear estado
311        self.esn.reset()
312        
313        # Procesar prompt
314        prompt_indices = self.tokenizer.encode(prompt)
315        
316        # Calentar el reservoir con el prompt
317        for idx in prompt_indices:
318            emb = self._get_embedding(idx)
319            self.esn._update_state(emb)
320        
321        # Generar
322        generated = list(prompt_indices)
323        bos_token = getattr(self.tokenizer, 'BOS_TOKEN', self.tokenizer.BOS)
324        current_idx = prompt_indices[-1] if len(prompt_indices) > 0 else self._get_token_id(bos_token)
325        
326        eos_token = getattr(self.tokenizer, 'EOS_TOKEN', self.tokenizer.EOS)
327        eos_idx = self._get_token_id(eos_token)
328        
329        for _ in range(max_tokens):
330            emb = self._get_embedding(current_idx)
331            state = self.esn._update_state(emb)
332            logits = np.dot(state, self.esn.W_out)
333            
334            if strategy == 'greedy':
335                next_idx = self._select_next_greedy(logits)
336            else:
337                next_idx = self._select_next_sampling(logits, temperature, top_k)
338            
339            next_idx = self._avoid_special_tokens(next_idx, logits)
340            generated.append(next_idx)
341            current_idx = next_idx
342            
343            if next_idx == eos_idx:
344                break
345        
346        return self.tokenizer.decode(np.array(generated))
347    
348    def get_stats(self) -> Dict:
349        """Estadísticas del modelo."""
350        memory_bytes = (
351            self.n_reservoir * self.embedding_dim * 8 +  # W_in
352            self.n_reservoir * self.n_reservoir * 8 +    # W_reservoir
353            self.n_reservoir * self.tokenizer.actual_vocab_size * 8  # W_out
354        )
355        
356        return {
357            'n_reservoir': self.n_reservoir,
358            'vocab_size': self.tokenizer.actual_vocab_size if hasattr(self.tokenizer, 'actual_vocab_size') else 0,
359            'embedding_dim': self.embedding_dim,
360            'is_trained': self.is_trained,
361            'memory_kb': memory_bytes / 1024
362        }

255    def _select_next_sampling(
256        self, 
257        logits: np.ndarray, 
258        temperature: float, 
259        top_k: int
260    ) -> int:
261        """Selección por sampling con top-k."""
262        import time
263        
264        # Aplicar temperatura
265        logits = logits / (temperature + 1e-8)
266        
267        # Top-k filtering
268        top_k_indices = np.argsort(logits)[-top_k:]
269        top_k_logits = logits[top_k_indices]
270        
271        # Softmax
272        exp_logits = np.exp(top_k_logits - np.max(top_k_logits))
273        probs = exp_logits / np.sum(exp_logits)
274        
275        # Samplear
276        rng = np.random.default_rng(int(time.time() * 1000) % (2**32))
277        chosen = rng.choice(len(top_k_indices), p=probs)
278        return int(top_k_indices[chosen])

280    def _avoid_special_tokens(self, next_idx: int, logits: np.ndarray) -> int:
281        """Evita seleccionar tokens especiales (PAD, UNK)."""
282        if next_idx in [0, 1]:
283            sorted_indices = np.argsort(logits)[::-1]
284            for idx in sorted_indices:
285                if idx not in [0, 1]:
286                    return int(idx)
287        return next_idx

304

320            self.esn._update_state(emb)
